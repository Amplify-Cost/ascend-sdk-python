@app.post("/mcp/actions/assess-risk-enterprise") 
async def assess_enterprise_risk(
    action_data: dict,
    current_user = Depends(get_current_user)
):
    """Enterprise risk assessment with framework-based methodology"""
    try:
        # Validate required fields
        required_fields = ["agent_id", "action_type", "resource"]
        for field in required_fields:
            if field not in action_data:
                raise HTTPException(status_code=400, detail=f"Missing required field: {field}")
        
        # Use enterprise framework for risk assessment
        risk_result = enterprise_risk_assessor.assess_action_risk(action_data)
        
        return {
            "action_data": action_data,
            "enterprise_risk_assessment": risk_result,
            "assessed_by": current_user.email if hasattr(current_user, 'email') else 'system',
            "assessment_timestamp": datetime.utcnow().isoformat(),
            "enterprise_benefits": {
                "methodology_credibility": "Industry-standard CVSS v3.1 + NIST 800-30 + MITRE ATT&CK",
                "audit_ready": "Complete documentation for SOC2 compliance", 
                "threat_intelligence": f"Mapped to MITRE tactic {risk_result.get('mitre_tactic_id')}"
            }
        }
    except HTTPException:
        raise
    except Exception as e:
        logger.error(f"Enterprise risk assessment failed: {str(e)}")
        raise HTTPException(status_code=500, detail=f"Risk assessment failed: {str(e)}")

@app.get("/enterprise/framework-status")
async def get_framework_status():
    """Enterprise framework status for monitoring and health checks"""
    try:
        # Test the framework with a sample assessment
        test_action = {
            "agent_id": "framework_health_check",
            "action_type": "test_assessment",
            "resource": "/health/test"
        }
        
        start_time = datetime.utcnow()
        test_result = enterprise_risk_assessor.assess_action_risk(test_action)
        end_time = datetime.utcnow()
        processing_time = (end_time - start_time).total_seconds()
        
        return {
            "framework_status": "operational",
            "version": enterprise_risk_assessor.framework_version,
            "last_tested": end_time.isoformat(),
            "performance": {
                "test_assessment_time": f"{processing_time:.4f} seconds",
                "performance_target": "< 1.0 seconds",
                "performance_acceptable": processing_time < 1.0
            },
            "frameworks_loaded": {
                "cvss_v3_1": "✅ Operational",
                "nist_800_30": "✅ Operational", 
                "mitre_attack": "✅ Operational"

        }
    except Exception as e:
        return {

            "error_message": str(e),
